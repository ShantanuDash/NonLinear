import numpy as np 
import matplotlib.pyplot as plt 
plt.style.use('dark_background')

def clifford_attractor(x, y, a=-2.1, b=-0.8, c=1.2, d=1.6):
    '''Returns the change in arguments x and y according to 
    the Clifford map equation. Kwargs a, b, c, and d are specified
    as constants.'''
    x_next = np.sin(a*y) + c*np.cos(a*x) 
    y_next = np.sin(b*x) + d*np.cos(b*y)
    return x_next, y_next

# number of iterations
iterations = 1000000

# initialization
X = np.zeros(iterations+1)
Y = np.zeros(iterations+1)

# starting point
X[0], Y[0] = 1, 1
# map
for i in range(iterations):
    x_next, y_next = clifford_attractor(X[i], Y[i])
    X[i+1] = x_next
    Y[i+1] = y_next 

# make and display figure.
plt.figure(figsize=(10, 10))
plt.plot(X, Y, '^', color='white', alpha = 0.5, markersize = 0.05)
plt.axis('on')
plt.show()

import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

# define equation
def rossler_attractor(x, y, z, a=0.1, b=0.1, c=14):
    x_dot = - y - z
    y_dot = x + a*y
    z_dot = b + z*(x - c)

    return x_dot, y_dot, z_dot
    
    # basic parameters
delta_t = 0.01 
steps = 100000

# initialize solutions arrays (+1 for initial conditions)
xx = np.empty((steps + 1))
yy = np.empty((steps + 1))
zz = np.empty((steps + 1))

# fill in initial conditions
xx[0], yy[0], zz[0] = (0.1, 0., 0.1)
# solve equation system
for i in range(steps):
    # Calculate derivatives
    x_dot, y_dot, z_dot = rossler_attractor(xx[i], yy[i], zz[i])
    
    xx[i + 1] = xx[i] + (x_dot * delta_t)
    yy[i + 1] = yy[i] + (y_dot * delta_t)
    zz[i + 1] = zz[i] + (z_dot * delta_t)
    
fig = plt.figure(figsize=(15,15))
ax=plt.axes(projection="3d")
plt.style.use('dark_background')

plt.plot(xx[:i], yy[:i], zz[:i],'-',color='white',lw=0.1)

# set limits
ax.set_xlim(-20,25)
ax.set_ylim(-25,20)
ax.set_zlim(0,50)
